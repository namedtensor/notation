Named Tensors: CNN / LeNet

Conv 2D

\[\begin{aligned} W &\in \mathbb{R}^{\mathsf{channel}:c,\mathsf{kh}:kh, \mathsf{kw}:kw} \\ h' &= h-kh +1, w' = w-kw +1 \\ b &\in \mathbb{R}^{\mathsf{height}:h', \mathsf{width}:w'} \\ \text{conv2d}(X; W, b) &= W \mathbin{\mathop{\boldsymbol\cdot}\limits_{\mathsf{channel, kh, kw}}} U + b \\ U &\in \mathbb{R}^{{\mathsf{channel}:c, \mathsf{height}:h', \mathsf{width}:w', \mathsf{kh}:kh, \mathsf{kw}:kw}} \\ U_{\mathsf{height}:i, \mathsf{width}:j, \mathsf{kh}:i', \mathsf{kw}:j'} &= X_{\mathsf{height}:i+i' - 1, \mathsf{width}:j+j' - 1} \end{aligned}\]

Max Pooling

\[\begin{aligned} X &\in \mathbb{R}^{\mathsf{height}:h, \mathsf{width}:w} \\ \text{maxpool2d}(X, kh, kw) &= \mathop{\text{max}}\limits_{\mathsf{kh, kw}} U \\ U &\in \mathbb{R}^{{\mathsf{height}:h / kh, \mathsf{width}:w / kw, \mathsf{kh}:kh, \mathsf{kw}:kw}} \\ U_{\mathsf{height}:i, \mathsf{width}:j, \mathsf{kh}:di, \mathsf{kw}:dj} & = X_{\mathsf{height}:i \times kh + di -1, \mathsf{width}:j \times kw + dj - 1} \end{aligned}\]

MLP

\[\begin{aligned} W^1 &\in \mathbb{R}^{\mathsf{inp}:i, \mathsf{hidden}:h}, W^2 \in \mathbb{R}^{ \mathsf{hidden}:h} \\ \text{MLP}(X; W, b) &= W^1 \mathbin{\mathop{\boldsymbol\cdot}\limits_{\mathsf{hidden}}} \text{ReLU}(W^1 \mathbin{\mathop{\boldsymbol\cdot}\limits_{\mathsf{inp}}} X + b^1) + b^2 \end{aligned}\]

Model

\[\begin{aligned} \textit{Params}: \\ W^1 & \in \mathbb{R}^{\mathsf{channels'}:c_2, \mathsf{channels}:c_1, \mathsf{kh}:kh_1, \mathsf{kw}:kw_1} \\ W^2 & \in \mathbb{R}^{\mathsf{channels'}:c_3, \mathsf{channels}:c_2, \mathsf{kh}:kh_2, \mathsf{kw}:kw_2} \\ U & \in \mathbb{R}^{\mathsf{in}:h_1, \mathsf{hidden}:h_2}, V \in \mathbb{R}^{\mathsf{hidden}:h_2, \mathsf{out}:h_3} \\ \\ \textit{Model}: \\ X^0 &\in \mathbb{R}^{\mathsf{batch}:b, \mathsf{channels}:1, \mathsf{height}:h, \mathsf{width}:w}\\ X^1 &= \left[\text{ReLU}(\text{conv2d}(X^0; W^1))\right]_{\mathsf{channels'}\rightarrow\mathsf{channels}} \\ X^2 &= \text{maxpool2d}(X^1, ph_1, ph_2) \\ X^3 &= \left[\text{ReLU}(\text{conv2d}(X^2; W^2))\right]_{\mathsf{channels'}\rightarrow\mathsf{channels}} \\ X^4 &= \text{maxpool2d}(X^3, ph_2, ph_2) \\ X^5 &\in \mathbb{R}^{\mathsf{batch}:b, \mathsf{inp}:h_1}\\ X^5_{\mathsf{inp}} &= X^4_{\mathsf{height}, \mathsf{width}, \mathsf{channels}}\\ O &= \mathop{\text{softmax}}\limits_{\mathsf{class}} \left[\text{MLP}(X^5; U, V)\right]_{\mathsf{out}\rightarrow\mathsf{class}}\end{aligned}\]
